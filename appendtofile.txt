
frontend haproxy-main
    bind *:80
    option forwardfor  
    default_backend apache_webservers    

backend apache_webservers
    balance roundrobin
    server websvr1	192.168.1.10:80 check
    server websvr2	192.168.1.11:80 check

listen stats
    bind :8800
    stats enable
    stats uri /
    stats hide-version
    stats auth admin:password
    default_backend apache_webservers
